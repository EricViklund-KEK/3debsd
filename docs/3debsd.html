354    def plot_grain(self, id: int) -> PolyData:
355        """
356        Generates a 3D visualization of a specific grain boundary.
357
358        Parameters
359        ----------
360        id : int
361            The unique integer identifier of the grain to be visualized.
362
363        Returns
364        -------
365        pv.PolyData

367            ready for interactive plotting or file export.
368
369        Examples
370        --------
371        >>> # Assuming 'mesh' is an initialized Mesh3D object
372        >>> grain_poly = mesh.plot_grain(id=0)
373        >>> grain_poly.plot(color='gold', show_edges=True) # doctest: +SKIP
374        
375        You can also save the resulting geometry to a file:
376        
377        >>> grain_poly.save("grain_0.vtk") # doctest: +SKIP
378        """
379        GB_vertices = [self._vor.ridge_vertices[face] for face in self._T_FG[:,[id]].tocoo().coords[0]]
380
381        # First, identify which vertices are actually used in GB_vertices
382        used_vertices = set()
383        for triangle in GB_vertices:
384            for idx in triangle:
385                used_vertices.add(idx)
386
387        # Create a mapping from old indices to new indices
388        old_to_new = {}
389        new_vertices = []
390
391        for i, idx in enumerate(sorted(used_vertices)):
392            old_to_new[idx] = i
393            new_vertices.append(self._vertices[idx])
394
395        # Convert vertices to numpy array
396        new_vertices = np.array(new_vertices)
397
398        # Update the indices in GB_vertices
399        new_GB_vertices = []
400        for triangle in GB_vertices:
401            new_triangle = [old_to_new[idx] for idx in triangle]
402            new_GB_vertices.append(new_triangle)
403
404        grain_mesh = pv.PolyData\
405            .from_irregular_faces(new_vertices, new_GB_vertices)\
406            .connectivity('largest')\
407            .triangulate()\
408            .clean()\
409            .compute_normals(auto_orient_normals=True)
410
411        polydata = grain_mesh
412
413        return polydata

320        point_ind = np.arange((len(vor.point_region)))
321        domain_ind = vor.point_region
322        T_PD[point_ind,domain_ind] = True
323
324        # Map grains to their phases
325        grain_phase = []
326        for grain_indices in grains:
327            point_ind = T_PD[:,grain_indices].nonzero()[0]
328            grain_phase.append(phase_flat[point_ind])
329
330        # Create a sparse matrix to represent the domain-grain relationships
331        T_DG = csr_array((len(vor.regions),len(grains)),dtype='bool')
332        grain_ind, domain_ind = np.array([[i,domain] for i, grain in enumerate(grains) for domain in grain]).T
333        T_DG[domain_ind,grain_ind] = True
334
335        # Compute the face-grain relationship
336        B_D = T_DG @ T_DG.T
337        B_D.eliminate_zeros()
338        subgrain_matrix = A_DGB * B_D
339        gb_matrix = A_DGB - subgrain_matrix
340        T_FG = (T_FD @ gb_matrix @ T_DG) * (T_FD @ T_DG)
341
342
343        self._vor = vor
344        self._T_FG = T_FG
345        self._vertices = vertices
346        self._grains = grains
347        self._grain_phase = grain_phase
